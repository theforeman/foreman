POST /api/config_templates
Create a template

Examples

POST /api/config_templates
{
  "config_template": {
    "template_kind_id": 1,
    "name": "RandomName",
    "template": "This is a test template"
  }
}
200
{
  "config_template": {
    "snippet": false,
    "template_combinations": [],
    "name": "RandomName",
    "audit_comment": null,
    "id": 1007981702,
    "template": "This is a test template",
    "": []
  }
}
