POST /api/config_templates
Create a template

Examples

POST /api/config_templates
{
  "config_template": {
    "template_kind_id": 1,
    "template": "This is a test template",
    "name": "RandomName"
  }
}
200
{
  "config_template": {
    "id": 1007981702,
    "template_combinations": [],
    "snippet": false,
    "": [],
    "template": "This is a test template",
    "audit_comment": null,
    "name": "RandomName"
  }
}
