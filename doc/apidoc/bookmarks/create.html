POST /api/bookmarks
Create a bookmark.

Examples

POST /api/bookmarks
{
  "bookmark": {
    "controller": "hosts",
    "public": false,
    "name": "foo-bar",
    "query": "bar"
  }
}
200
{
  "bookmark": {
    "controller": "hosts",
    "owner_id": 135138680,
    "public": false,
    "owner_type": "User",
    "name": "foo-bar",
    "id": 980190963,
    "user": {
      "email": "admin@someware.com",
      "name": "Admin User",
      "login": "admin"
    },
    "query": "bar"
  }
}
