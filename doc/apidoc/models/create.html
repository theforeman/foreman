POST /api/models
Create a model.

Examples

POST /api/models
{
  "model": {
    "name": "new model"
  }
}
201
{
  "model": {
    "hardware_model": null,
    "created_at": "2012-11-13T13:49:24Z",
    "vendor_class": null,
    "updated_at": "2012-11-13T13:49:24Z",
    "name": "new model",
    "id": 980190963,
    "info": null
  }
}
